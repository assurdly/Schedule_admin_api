const response = pm.response.json();

pm.test("Validate response status", function () {
    pm.response.to.have.status(200);
});


// Validate response structure
pm.test("Validate response structure", function () {
    pm.expect(response).to.have.property('message');
    pm.expect(response).to.have.property('data').and.not.empty;
});

// Validate response data types
pm.test("Validate response data types", function () {
    pm.expect(response.status).to.be.a('string');
    pm.expect(response.message).to.be.a('string');
    pm.expect(response.data).to.be.an('object');
});
