const jsonData = pm.response.json();

pm.test('response code should be 404', function () {
 pm.response.to.have.status(404);
});

pm.test('Validate response structure', function () {
 pm.expect(jsonData).to.be.an('object');
 pm.expect(jsonData).to.have.property('status').that.equals('error');
 pm.expect(jsonData).to.have.property('message').that.equals('Promotion not found');
});
