const jsonData = pm.response.json();

pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Response returns valid message", function () {
    pm.expect(jsonData.message).to.eql('Aggregator name already exists. Please choose a different name.')
})

pm.test("Data returns empty", function () {
    pm.expect(jsonData.data).to.eql({}).and.be.empty
})
