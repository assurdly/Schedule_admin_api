const jsonData = pm.response.json();

pm.test("Status code is 400", function () {
    pm.response.to.have.status(400);
});

pm.test("Valid error message is returned", function () {
    pm.expect(jsonData.message).to.eql("Base URL cannot be empty");
});

pm.test("No data is returned", function () {
    pm.expect(jsonData.data).to.eql({}).and.be.empty;
});
