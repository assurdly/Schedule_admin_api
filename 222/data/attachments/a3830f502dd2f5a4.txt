const jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test('Validate data object', function () {
    pm.expect(jsonData.data).to.have.property('id');
    pm.expect(jsonData.data).to.have.property('name');
    pm.expect(jsonData.data).to.have.property('description');
    pm.expect(jsonData.data).to.have.property('product_group_id');
    pm.expect(jsonData.data).to.have.property('product_id');
    pm.expect(jsonData.data).to.have.property('upload_result');
    pm.expect(jsonData.data).to.have.property('upload_status');
    pm.expect(jsonData.data).to.have.property('stats');
    pm.expect(jsonData.data.stats).to.be.an('object');
})
